from pyspark.sql import SparkSession
from pyspark.sql.functions import from_json, col, window, count
from pyspark.sql.types import StructType, StructField, IntegerType, StringType, FloatType
import logging

# Crear sesión Spark
spark = SparkSession.builder \
    .appName("CompraventasStreaming") \
    .getOrCreate()

spark.sparkContext.setLogLevel("WARN")

# --- Definir el esquema del JSON recibido ---
schema = StructType([
    StructField("ANO", IntegerType()),
    StructField("COORDENADA_X", FloatType()),
    StructField("COORDENADA_Y", FloatType()),
    StructField("ID_COMPRAVENTA", IntegerType()),
    StructField("OBJECTID", IntegerType()),
    StructField("point", StringType())
])

# --- Leer desde Kafka ---
df = spark.readStream \
    .format("kafka") \
    .option("kafka.bootstrap.servers", "localhost:9092") \
    .option("subscribe", "compraventas") \
    .load()

# --- Parsear los mensajes JSON ---
parsed_df = df.select(from_json(col("value").cast("string"), schema).alias("data")).select("data.*")

# --- Análisis: contar cantidad de compraventas por año ---
conteo_ano = parsed_df.groupBy("ano").agg(count("id_compraventa").alias("total_compraventas"))

# --- Mostrar resultados en tiempo real ---
query = conteo_ano.writeStream \
    .outputMode("complete") \
    .format("console") \
    .start()

query.awaitTermination()
